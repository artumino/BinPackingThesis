\begin{algorithm}[H] \label{algo:sp_bestinsertion}
    \DontPrintSemicolon
    \SetAlgoLined
    \SetKwInOut{Input}{input}
    \SetKwInOut{Output}{output}
    \Input{$Z, I$}
    \Output{$p$}
    $p \leftarrow \text{none}$\;
    \ForAll{$(z, I_{support}, I_{upper}) \in Z$}{
        \ForAll{possible coordinate changes}{
            \CommentSty{//Initialize reference points}
            $refPoints \gets \emptyset$\;
            \If{$z = 0$}{
                $refPoints \gets (0,0)$\;
            }
            \ForAll{$i \in I_{support}$}{
                $refPoints \gets refPoints \cup {(x_i, y_i)}$\;
            }
        }
        \If{$p \neq \text{none}$}{
            \Return{$p$}
        }
    }
    \Return{$p$}
    \caption{SP Best Insertion}
\end{algorithm}
